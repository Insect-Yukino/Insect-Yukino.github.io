class Solution {
    public int lengthOfLongestSubstring(String s) {
        Set<Character> window = new HashSet<>(); // 当前窗口内出现过的字符
        int n = s.length();
        int left = 0;       // 窗口左边界（包含）
        int right = 0;      // 窗口右边界（不包含）
        int ans = 0;

        while (right < n) {
            char c = s.charAt(right);

            // 如果 c 已经在窗口里，说明重复：不断移动 left 缩小窗口，直到 c 不重复
            while (window.contains(c)) {
                window.remove(s.charAt(left));
                left++;
            }

            // 现在 c 不重复了，可以加入窗口并扩展 right
            window.add(c);
            right++;

            // 更新答案：窗口长度 = right - left
            ans = Math.max(ans, right - left);
        }

        return ans;
    }
}
